# Calculate portfolio returns and volatility
portfolio_returns = []
portfolio_volatility = []
all_weights = []
position_status = [] # To store "Shorts Allowed" or "No Shorts"

num_trading_days = 252 # Number of trading days in a year

for single_weights in weights:
    all_weights.append(single_weights)
    mean_returns_np = np.array(mean_returns)
    returns = np.dot(single_weights, mean_returns_np) * num_trading_days # Annualize returns
    portfolio_returns.append(returns)

    covariance_matrix_np = np.array(covariance_matrix)
    volatility = np.sqrt(single_weights.T @ covariance_matrix_np @ single_weights) * np.sqrt(num_trading_days) # Annualize volatility
    portfolio_volatility.append(volatility)

    position_status.append("Shorts Allowed" if shortsOK else "No Shorts")
